    Enlarge the Colorpicker size

    This example shows how to tweak the Colorpicker styles and settings to make it look bigger.

    Code
    <input id="cp1" type="text" class="form-control input-lg" value="#6D2781"/>
<style>
  .colorpicker-2x .colorpicker-saturation {
    width: 200px;
    height: 200px;
  }

  .colorpicker-2x .colorpicker-hue,
  .colorpicker-2x .colorpicker-alpha {
    width: 30px;
    height: 200px;
  }

  .colorpicker-2x .colorpicker-color,
  .colorpicker-2x .colorpicker-color div {
    height: 30px;
  }

  .colorpicker-saturation .colorpicker-guide,
  .colorpicker-saturation .colorpicker-guide b {
    height: 10px;
    width: 10px;
    border-radius: 10px;
  }
</style>
<script>
  $(function () {
    $('#cp1').colorpicker({
      customClass: 'colorpicker-2x',
      sliders: {
        saturation: {
          maxLeft: 200,
          maxTop: 200
        },
        hue: {
          maxTop: 200
        },
        alpha: {
          maxTop: 200
        }
      }
    });
  });
</script>
